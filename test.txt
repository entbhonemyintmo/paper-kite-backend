my project folder structure is

project-root/
  │ └──src/
  │      │
  │      ├── config/
  │      │   │   
  │      │   ├── database.js         // Database configuration file
  │      │   └── env.js              // Environment variables configuration
  │      │
  │      ├── controllers/
  │      │   └── yourController.js   // Controllers for handling business logic
  │      │
  │      ├── middleware/
  │      │   └── authMiddleware.js   // Middleware functions (e.g., authentication)
  │      │
  │      ├── models/
  │      │   └── yourModel.js        // Mongoose models for interacting with MongoDB
  │      │
  │      ├── routes/
  │      │   └── yourRoutes.js       // Route definitions and middleware usage
  │      │
  │      ├── services/
  │      │   └── yourService.js      // Additional services (e.g., external API calls)
  │      │
  │      ├── utils/
  │      │   └── helperFunctions.js  // Utility/helper functions
  │      │
  │      ├── index.js                  // Main application file where Express is configured
  │      
  ├── package.json            // Project dependencies and scripts
  └── .env                    // Environment variables (ignored in version control)
